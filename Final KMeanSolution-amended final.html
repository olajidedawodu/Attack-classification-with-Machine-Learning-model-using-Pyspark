#k means determine k
cost = []
K = range(1,10)
for k in K:
    kmeanModel = KMeans(n_clusters=k).fit(cluster_final_data)
    kmeanModel.fit(cluster_final_data)
    cost.append(sum(np.min(cdist(cluster_final_data, kmeanModel.cluster_centers_, 'euclidean'), axis=1)) / cluster_final_data.shape[0])

# Plot the elbow
plt.plot(K, cost, 'bx-')
plt.xlabel('k')
plt.ylabel('cost')
plt.title('The Elbow Method showing the optimal value k')
